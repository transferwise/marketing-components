export function ColorToken(props) {
  return (
    <div className="d-flex align-items-center m-b-2">
      <div
        style={{
          backgroundColor: `var(${props.token})`,
          border: 'solid 1px rgba(0, 0, 0, 0.1)',
          borderRadius: '100%',
          width: '48px',
          height: '48px',
        }}
      />
      <div className="m-l-2">{props.token}</div>
    </div>
  );
}

**Note:** To use these properties in your application, you need `neptune-tokens.css` or `neptune-core.css` on your page. Tokens are exposed on `:root`. Refer to [Installing CSS](/about/Setup#installing-css) on the Setup page to get started.
